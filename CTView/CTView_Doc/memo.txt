//	MAGADBG --  コンパイラ定義  MAGADBG
//                  ライブラリ      magadbg.lib
//

メッセージ出力関数，MAGADBG（VC++のMFC用）対応  
	void  println_message(char* fmt, ...);


ランタイムで制御可能(bool DeBugMode) なメッセージ出力関数，MAGADBG（VC++のMFC用）対応  
	void  println_debug_message(char* fmt, ...);


コンパイル時(_DEBUG) の組み込みデバッグ用のメッセージ表示には以下のマクロを使用すること　
	DEBUG_Info(char* fmt, ...);
	DEBUG_Warning(char* fmt, ...);
	DEBUG_Error(char* fmt, ...);
